#include <stdio.h>

int main(){
  int num[3][4];
  int soma;
  int i;
  int j;
  
  for(i=0;i<3;i++){
    for(j=0;j<4;j++){
      printf("[%d] [%d] da matriz..:",i,j);
      scanf("%d", &num[i][j]);
    }
  }
  
  for(i=0;i<3;i++){
    for(j=0;j<4;j++){
        printf("[%d] ",num[i][j]);
    }
    printf("\n");
  }
  
  for(j=0;j<4;j++){
    soma = 0;
    for(i=0;i<3;i++){
        soma+=num[i][j];    
    }
    printf("A soma da coluna %d e igual..%d\n", j, soma);
  }


  return 0;
}
