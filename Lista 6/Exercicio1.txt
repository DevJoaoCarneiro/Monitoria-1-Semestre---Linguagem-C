  Leia uma matriz 3x3 de n√∫meros inteiros;

  Exiba os elementos da matriz na forma matricial. */

#include <stdio.h>

int main(){
  int num[3][3];
  int i;
  int j;
  
  for(i=0;i<3;i++){
    for(j=0;j<3;j++){
      printf("[%d] [%d] da matriz..:",i,j);
      scanf("%d", &num[i][j]);
    }
  }
  
  for(i=0;i<3;i++){
    for(j=0;j<3;j++){
      printf("[%2d] ",num[i][j]);
    }
    printf("\n");
  }

  return 0;
}
